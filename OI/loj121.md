> 
题目描述
这是一道被离线爆艹的模板题。
你要维护一张无向简单图。你被要求加入删除一条边及查询两个点是否连通。
*0：加入一条边。保证它不存在。
*1：删除一条边。保证它存在。
*2：查询两个点是否联通。
输入格式
输入的第一行是两个数 N M。N≤5000,M≤500000。
接下来 M 行，每一行三个数 op x y。op 表示操作编号。
输出格式
对于每一个 op=2 的询问，输出一行 Y 或 N ，表示两个节点是否连通。

　　离线读取每条边删除的时间，维护生成树。
　　如果现在加入一条新边，而且边的两个端点在同一颗树上的话，把加边之后构成的环上删除时间最早的边删了，因为直到它退出之前，在这个路经上的其他点能代替它形成联通块。

```C++
#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=5005;
const int M=500005;
int n,m,cnt,st[N][N],cx[2][M],ed[M],map[M],fmp[M];
//map:node->edge
bool vis[M];//edge
struct node{
	int fa,mn,mp,ed,s[2];
	bool rev;
}tr[N+M];
struct edge{
	int a,b,ed;
}eg[M];
inline int nxi(){
	int x=0;
	char c;
	while((c=getchar())>'9'||c<'0');
	for(;c>='0'&&c<='9';c=getchar()) x=x*10+c-'0';
	return x;
}
inline void gmn(int x,int f){
	if(tr[x].mn<tr[f].mn){
		tr[f].mn=tr[x].mn;
		tr[f].mp=tr[x].mp;
	}
}
inline void upd(int x){
	tr[x].mn=tr[x].ed;
	tr[x].mp=x;
	gmn(tr[x].s[0],x);
	gmn(tr[x].s[1],x);
}
inline bool isrt(int x){
	return tr[tr[x].fa].s[0]!=x&&tr[tr[x].fa].s[1]!=x;
}
inline bool son(int x){
	return tr[tr[x].fa].s[1]==x;
}
void psh(int x){
	if(!isrt(x)) psh(tr[x].fa);
	if(tr[x].rev){
		tr[tr[x].s[0]].rev^=1;
		tr[tr[x].s[1]].rev^=1;
		swap(tr[x].s[0],tr[x].s[1]);
		tr[x].rev=0;
	}
}
inline void rot(int x){
	bool k=son(x);
	int f=tr[x].fa,ff=tr[f].fa,&c=tr[x].s[k^1];
	if(!isrt(f)) tr[ff].s[son(f)]=x;
	tr[x].fa=ff;
	tr[f].s[k]=c;
	tr[f].fa=x;
	if(c) tr[c].fa=f;
	c=f;
	upd(f);
	upd(x);
}
inline void splay(int x){
	psh(x);
	while(!isrt(x)){
		int f=tr[x].fa;
		if(!isrt(f)){
			rot(son(f)==son(x)?f:x);
		}
		rot(x);
	}
}
inline void acs(int x){
	int y=0;
	while(x){
		splay(x);
		tr[x].s[1]=y;
		upd(x);
		x=tr[y=x].fa;
	}
}
inline void mrt(int x){
	acs(x);
	splay(x);
	tr[x].rev^=1;
}
inline void link(int e){//edge
	map[++cnt]=e;
	fmp[e]=cnt;
	vis[e]=1;
	int x=eg[e].a,y=eg[e].b;
	tr[cnt]=(node){y,eg[e].ed,cnt,eg[e].ed,0,0};
	mrt(x);
	tr[x].fa=cnt;
}
inline void cut(int x,int y){
	mrt(x);
	acs(y);
	splay(y);
	tr[y].s[0]=tr[x].fa=0;
	upd(y);
}
inline void del(int p){//node p
	int tp=map[p];
	cut(eg[tp].a,p);
	cut(p,eg[tp].b);
	vis[tp]=0;
}
inline bool ask(int a,int b){
	mrt(a);
	acs(b);
	splay(b);
	while(tr[b].s[0]) b=tr[b].s[0];
	return a==b;
}

int main(){
	n=nxi(),m=nxi();
	for(int i=0;i<=n;++i) tr[i].mn=tr[i].ed=M;
	cnt=n;
	for(int op,a,b,i=1;i<=m;++i){
		op=nxi(),a=nxi(),b=nxi();
		switch(op){
			case 0:
				st[a][b]=st[b][a]=i;
				eg[i]=(edge){a,b,M};
				break;
			case 1:
				eg[st[a][b]].ed=i;
				ed[i]=st[a][b];
				break;
			case 2:
				cx[0][i]=a;
				cx[1][i]=b;
		}
	}
	for(int i=1;i<=m;++i){
		if(cx[0][i]){
			puts(ask(cx[0][i],cx[1][i])?"Y":"N");
		}else if(ed[i]){
			if(vis[ed[i]]) del(fmp[ed[i]]);
		}else{
			int a=eg[i].a,b=eg[i].b;
			if(!ask(a,b)) link(i);
			else{
				mrt(a);
				acs(b);
				//splay(b)?
				if(tr[b].mn<eg[i].ed){
					int p=tr[b].mp;
					del(p);
					link(i);
				}
			}
		}
	}
	return 0;
}
```
